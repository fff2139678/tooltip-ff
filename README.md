## Install

```
npm i tooltip-ff
```

## Usage

```
import React from 'react';
import ReactDOM from 'react-dom';

import XToolTip from 'tooltip-ff';

ReactDOM.render (
    <XToolTip position={'right'} content={'测试'}>
      {'我是测试按钮'}
    </XToolTip>,
  document.getElementById ('app')
);

```
